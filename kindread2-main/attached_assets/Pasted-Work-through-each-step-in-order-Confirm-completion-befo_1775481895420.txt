The notifications table already exists. Add these 3 Supabase 
database triggers via SQL:

  -- Trigger 1: Like notification
  CREATE OR REPLACE FUNCTION notify_on_like()
  RETURNS TRIGGER AS $$
  DECLARE
    bookmark_owner UUID;
  BEGIN
    SELECT user_id INTO bookmark_owner 
    FROM bookmarks WHERE id = NEW.bookmark_id;
    
    IF bookmark_owner != NEW.user_id THEN
      INSERT INTO notifications 
        (user_id, actor_id, type, bookmark_id)
      VALUES 
        (bookmark_owner, NEW.user_id, 'like', NEW.bookmark_id);
    END IF;
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;

  CREATE TRIGGER trigger_notify_like
    AFTER INSERT ON bookmark_likes
    FOR EACH ROW EXECUTE FUNCTION notify_on_like();

  -- Trigger 2: Comment notification
  CREATE OR REPLACE FUNCTION notify_on_comment()
  RETURNS TRIGGER AS $$
  DECLARE
    bookmark_owner UUID;
  BEGIN
    SELECT user_id INTO bookmark_owner 
    FROM bookmarks WHERE id = NEW.bookmark_id;
    
    IF bookmark_owner != NEW.user_id THEN
      INSERT INTO notifications
        (user_id, actor_id, type, bookmark_id)
      VALUES
        (bookmark_owner, NEW.user_id, 'comment', NEW.bookmark_id);
    END IF;
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;

  CREATE TRIGGER trigger_notify_comment
    AFTER INSERT ON bookmark_comments
    FOR EACH ROW EXECUTE FUNCTION notify_on_comment();

  -- Trigger 3: Session join notification
  CREATE OR REPLACE FUNCTION notify_on_session_join()
  RETURNS TRIGGER AS $$
  DECLARE
    club_host UUID;
  BEGIN
    SELECT host_id INTO club_host 
    FROM clubs WHERE id = NEW.club_id;
    
    IF club_host != NEW.user_id THEN
      INSERT INTO notifications
        (user_id, actor_id, type, club_id)
      VALUES
        (club_host, NEW.user_id, 'session_join', NEW.club_id);
    END IF;
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;

  CREATE TRIGGER trigger_notify_session_join
    AFTER INSERT ON club_members
    FOR EACH ROW EXECUTE FUNCTION notify_on_session_join();
